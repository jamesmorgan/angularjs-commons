Filters common filters for any application

Capitalise

Example:

{{ mc.capitalised.value | capitalise }}

Usage:

{{ myText | capitalise }}

Truncate

Example:

{{ mc.truncate.value | truncate:10 }}

Usage:

{{ myText | truncate:25:"..." }}

To Lower

Example:

{{ mc.toLower.value | toLower }}

Usage:

{{ myText | toLower }}

To Upper

Example:

{{ mc.toUpper.value | toUpper }}

Usage:

{{ myText | toUpper }}

To Camel Case

Example:

{{ mc.toCamel.value | toCamel }}

Usage:

{{ myText | toCamel }}

Slice

Example:

{{ mc.slice.value | slice:0:mc.slice.to }}

Usage:

{{ model.value | slice:0:2 }}

Clean & Capitalise

Useful for converting ENUM values

Example:

{{ mc.cleanAndCapitalised.value | cleanAndCapitalise }}

Usage:

{{ myText | cleanAndCapitalise }}

Reverse

Useful for changing UI logic without changing API

Example:

{{ mc.reverse.value | reverse }}

Usage:

{{ model.value | reverse }}
